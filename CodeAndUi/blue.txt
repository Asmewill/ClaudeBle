    # 开启事件循环
    def start_async_loop(self):
        asyncio.set_event_loop(self.loop)
        self.loop.run_forever()

    def update_status(self, message):
        self.status_label.config(text=message)


    async def handle_notifications(self, sender, data):
        try:
            if sender.uuid == self.data_uuid:
                self.data_text.insert(tk.END, f"收到数据: {data}\n")
            elif sender.uuid == self.heartbeat_uuid:
                # 解码数据
                data_str = data.decode("utf-8")
                #print(f"接收到的数据：{data_str}")
                try:
                    # 解析 "标志值,流速" 格式的字符串
                    parts = data_str.split(',')
                    int_part = int(parts[0])  # 标志值
                    float_part = float(parts[1])  # 流速值，单位 L/min
                except (ValueError, IndexError) as e:
                    print(f"解析错误：{e}")
                    return
                value = int_part  # 解析返回值

                # 分析位图，提取所有设置的标志位对应的名称
                active_flags = []
                for bit, name in self.CONTROLFUNCTION.items():
                    if value & bit:  # 检查该标志位是否被设置
                        active_flags.append(name)

                # 将结果拼接成字符串
                flags_str = "\n\t".join(active_flags) if active_flags else "NUL"

                # 获取当前时间戳
                current_time = time.time()

                # 计算时间间隔
                if self.last_timestamp is not None:
                    time_delta = current_time - self.last_timestamp  # 单位：秒
                    float_part = float_part * 0.4
                    # 流量累积计算 (流速/分钟 × 秒数)
                    flow_this_period = (float_part / 60.0) * time_delta
                    self.total_flow += flow_this_period

                    # 更新 GUI 界面
                    self.heartbeat_text.insert(
                        tk.END,
                        f"收到返回值:\n"
                        f"\t{flags_str}\n"
                        f"\t流速: {float_part:.6f} L/Min\n"
                        f"\t本周期流量: {flow_this_period:.6f} L\n"
                        f"\t总流量: {self.total_flow:.6f} L\n"
                    )
                    self.heartbeat_text.see(tk.END)

                # 更新最新时间戳
                self.last_timestamp = current_time
        except Exception as e:
            self.heartbeat_text.insert(tk.END, f"通知处理错误: {str(e)}\n")


    def connect_device(self):
        if not self.selected_device:
            messagebox.showerror("错误", "请先选择设备！")
            return

        address = self.selected_device.split("[")[1].split("]")[0]

        async def connect():
            try:
                self.ble_client = BleakClient(address)
                await self.ble_client.connect()
                if self.ble_client.is_connected:
                    self.total_flow = 0.0
                    self.update_status(f"状态：已连接到 {self.selected_device}")
                    #await self.ble_client.start_notify(self.data_uuid, self.handle_notifications)
                    await self.ble_client.start_notify(self.heartbeat_uuid, self.handle_notifications)
                    self.keep_heartbeat = True
                    asyncio.create_task(self.heartbeat_loop())  # 自动启动心跳任务
            except Exception as e:
                messagebox.showerror("错误", f"连接失败：{str(e)}")

        asyncio.run_coroutine_threadsafe(connect(), self.loop)

    def disconnect_device(self):
        if not self.ble_client or not self.ble_client.is_connected:
            messagebox.showerror("错误", "没有设备连接！")
            return

        async def disconnect():
            try:
                self.keep_heartbeat = False  # 停止心跳任务
                #await self.ble_client.stop_notify(self.data_uuid)
                await self.ble_client.stop_notify(self.heartbeat_uuid)
                await self.ble_client.disconnect()
                self.update_status("状态：未连接")
                self.ble_client = None
            except Exception as e:
                messagebox.showerror("错误", f"断开连接失败：{str(e)}")

        asyncio.run_coroutine_threadsafe(disconnect(), self.loop)

    # 扫描设备
    def scan_devices(self, timeout=5):
        async def scan():
            self.devices_box.delete(0, tk.END)
            self.update_status("扫描中，请稍等...")
            try:
                devices = await BleakScanner.discover(timeout=timeout)
                filtered_name = self.device_name_filter.get().strip()

                if filtered_name:
                    devices = [d for d in devices if filtered_name.lower() in (d.name or "").lower()]

                devices.sort(key=lambda d: d.rssi or -100, reverse=True)

                for device in devices:
                    self.devices_box.insert(
                        tk.END, f"{device.name or 'Unknown'} [{device.address}], RSSI: {device.rssi}"
                    )

                if not devices:
                    self.update_status("未找到符合条件的设备")
                else:
                    self.update_status("扫描完成")
            except Exception as e:
                self.update_status("扫描出现错误！")
                messagebox.showerror("错误", f"扫描失败：{str(e)}")

        asyncio.run_coroutine_threadsafe(scan(), self.loop)

    def on_scan_button_click(self):
        try:
            timeout = int(self.timeout_entry.get())
            if timeout <= 0:
                raise ValueError("超时时间必须是正数")
        except ValueError:
            messagebox.showerror("错误", "请输入有效的扫描超时时间！")
            return

        self.scan_devices(timeout=timeout)

    def on_select_device(self, event):
        self.selected_device = self.devices_box.get(tk.ACTIVE)

        # 添加 send_control_message 方法，用于发送控制消息
    def send_control_message(self, control_function):
        """发送控制消息到设备"""
        if not self.ble_client or not self.ble_client.is_connected:
            messagebox.showerror("错误", "设备未连接，无法发送控制消息！")
            return

        # 检查 data_uuid 是否已设置
        if not self.data_uuid:
            messagebox.showerror("错误", "特性 UUID 未设置！")
            return

        async def send_message():
            try:
                # 将控制功能值发送给设备
                # 将控制功能值表示为字符串
                string_data = str(control_function)

                # 将字符串编码为字节数据（按 UTF-8 编码）
                data = string_data.encode('utf-8')
                if self.ble_client.is_connected:
                    await self.ble_client.write_gatt_char(self.data_uuid, data)

                    # 在数据收发窗口显示已发送的消息内容
                    self.data_text.insert(tk.END, f"已发送控制消息: {control_function}\n")
                    self.data_text.see(tk.END)
            except Exception as e:
                messagebox.showerror("错误", f"发送控制消息失败：{str(e)}")

        asyncio.run_coroutine_threadsafe(send_message(), self.loop)


    # 周期性发送心跳
    async def heartbeat_loop(self):
        try:
            # 将控制功能值发送给设备
            # 将控制功能值表示为字符串
            string_data = str(1)

            # 将字符串编码为字节数据（按 UTF-8 编码）
            data = string_data.encode('utf-8')
            while self.keep_heartbeat and self.ble_client.is_connected:
                await self.ble_client.write_gatt_char(self.heartbeat_uuid, data)
                self.heartbeat_text.insert(tk.END, "已发送心跳: 1\n")
                self.heartbeat_text.see(tk.END)
                await asyncio.sleep(1)
        except Exception as e:
            self.heartbeat_text.insert(tk.END, f"心跳任务出错: {str(e)}\n")
        finally:
            self.keep_heartbeat = False


# 启动程序
if __name__ == "__main__":
    root = tk.Tk()
    app = BLEApplication(root)
    root.mainloop()
